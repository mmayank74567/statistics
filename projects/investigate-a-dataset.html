successful_movies = tmdb_data[tmdb_data['Profit'] >= 40000000]
def success(column):
    data_list = successful_movies[column].str.cat(sep = '|')
    data_list = pd.Series(data_list.split('|'))
    counter = data_list.value_counts(ascending = False)
    return counter
